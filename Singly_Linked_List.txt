#include<bits/stdc++.h>
using namespace std;

class node {
public :
int val ;
node * next;
};
node * head = NULL;
void create()
{
   cout<<"enter the number of values to insert"<<endl;
   int n,mj;
   cin>>n;
   cout<<"enter the values"<<endl;
   node * p ;
   for(int i=0;i<n;i++)
   {
     cin>>mj;
     if(head == NULL)
     {
       node* temp=new node();
       temp->val =mj;
       temp->next = NULL;
       head = temp;
       p = temp;
     }
     else
     {
       node* temp=new node();
       temp->val =mj;
       temp->next = NULL;
       p->next = temp;
       p = p->next;
     }
   }
}

void show()
{
  node * p = head;
  while(p!=NULL)
  {
    cout<<p->val<<" ";
    p = p->next;
  }
  cout<<endl;
}
void update()
{
  node * p = head;
  while(p!=NULL)
  {
    if(p->val %2 == 0)
     p->val = 0;
     p = p->next;
  } 
}
int length()
{
  node * p = head;
  int mj = 0;
  while(p!=NULL)
  {
      mj++;
     p = p->next;
  }
  return mj; 
}
void insert()
{
  int j,pos;
  cout<<"enter the number to insert and position to insert "<<endl;
  cin>>j>>pos;
  int mjj = length();
  //cout<<mjj<<endl;
  if(pos<1 || pos>mjj+1)
  {cout<<"enter a valid position "<<endl;}
  else
  {
     if(pos == 1)
      {
       // cout<<"hi"<<endl;
        node* temp = new node();
        temp->next = head;
        temp->val = j;
        head = temp;
      }
     else
     {
         node * p = head;
         pos-=2;
         while(pos--)
          p = p->next;
          node * a = p;
          node * b = p->next;
          node* temp= new node();
          temp->next = b;
          temp->val = j;
          a->next = temp;
     }
  }
 

}

void deletee()
{
  cout<<"enter the position to delete"<<endl;
  int pos;
 /// cout<<"enter the number to insert and position to insert "<<endl;

  //cout<<mjj<<endl;

  {cout<<"enter a valid position "<<endl;}
  else
  {
    if(pos == 1)
      {
       // cout<<"hi"<<endl;
        node* temp = head;
        head = head->next;
        delete temp;
      }
      else
     {
         node * p = head;
         pos-=2;
         while(pos--)
          p = p->next;
          node * a = p;
          node * b = p->next;
          node *c = b->next;
          delete b;
          a->next=c;
      }
  }
}
